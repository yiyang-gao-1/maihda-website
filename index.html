                <h2>Welcome to the MAIHDA Resource Hub</h2>

                <p class="lead">The <strong>Multilevel Analysis of Individual Heterogeneity and Discriminatory Accuracy (MAIHDA)</strong> approach is an innovative intersectional methodology for understanding health and social inequalities.</p>

                

                <h3 class="mt-4">What is MAIHDA?</h3>

                <p>MAIHDA is a quantitative intersectional approach that:</p>

                <ul>

                    <li>Examines how multiple social identities combine to shape outcomes</li>

                    <li>Uses multilevel models to partition variance between individuals and social strata</li>

                    <li>Provides measures of discriminatory accuracy for understanding inequality</li>

                    <li>Offers a framework for policy-relevant intersectional analysis</li>

                </ul>

        // Download BibTeX function

        function downloadBibTeX() {

            fetch('https://yiyang-gao-1.github.io/maihda-website/data/papers.json')

                .then(response => response.json())

                .then(papers => {

                    let bibtex = '';

                    for (const [id, paper] of Object.entries(papers)) {

                        bibtex += `@article{${id},

  title={${paper.title}},

  author={${paper.authors}},

  year={${paper.year}},

  journal={${paper.journal}}

}\n\n`;

                    }

                    const blob = new Blob([bibtex], { type: 'text/plain' });

                    const url = window.URL.createObjectURL(blob);

                    const a = document.createElement('a');

                    a.href = url;

                    a.download = 'maihda_papers.bib';

                    a.click();

                });

        }

    </script>
